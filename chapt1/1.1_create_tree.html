公式¶

  • ,$D'$
$$ d=\sqrt{(D+D')^2+x^2} $$
  • , :
$$ \begin{align} TOF_n(ns)&= \frac{72.29824 \cdot d(m)}{\sqrt{E_n(MeV)}} \\ \gammaTOF_{\gamma}(ns)&= 3.333\cdot d(m) \end{align} $$
  • (offset)
$$ \begin{align} t_u &= TOF+(L-x)/v_{sc}\\ t_d &= TOF+(L+x)/v_{sc}\\ x &= (t_d-t_u)*v_{sc}/2\\ TOF&=(t_u+t_d)/2 -L/v_{sc}\\ \end{align} $$
  • (PMToffset)
$$ \begin{equation} t_u=TOF+(L-x)/v_{sc}+t_{u\_off}\\ t_d=TOF+(L+x)/v_{sc}+t_{d\_off}\\ x=(t_d-t_u)*v_{sc}/2+C_x\\ TOF=(t_u+t_d)/2 +C_{tof}\\ \end{equation} $$$$ \begin{align} C_x&=(t_{u\_off}-t_{d\_off})*v_{sc}/2\\ C_{tof}&=-L/v_{sc}+(t_{u\_off}-t_{d\_off})/2 \end{align} $$
  • x $2Q_0$:
$$ Q_u=Q_0 e^{-(L-x)/\lambda} \\ Q_d=Q_0 e^{-(L+x)/\lambda}\\ x=\frac{\lambda}{2}ln\frac{Q_u}{Q_d}\\ Q_0^2=e^{2L/\lambda}*Q_u*Q_d\\ $$

:

  • (FWHM)$T_{res.}$
  • $\Delta E/E$: $E_{res.}$

  • TRandom3 class
    • Uniform(Double_t x1, Double_t x2);
    • Gaus(Double_t mean, Double_t sigma)

代码¶

// 
// tree.cc
// ROOT
// root -l //ROOT
// .x tree.cc //
// .q //root
// root -l tree.root //ROOT
void tree(){ 
//
  const Double_t D = 500.;//cm, distance between target and the scin.(Center)
  const Double_t L = 100.;//cm, half length of the scin.
  const Double_t dD = 5.;//cm, thickness of the scin.
  const Double_t TRes = 1.;//ns, time resolution(FWHM) of the scintillator.
  const Double_t Lambda = 380.;//cm, attenuation lenght of the scin.
  const Double_t QRes = 0.1;//relative energy resolution(FWHM) of the scin. 
  const Double_t Vsc = 7.5;//ns/cm, speed of light in the scin.
  const Double_t En0 = 100;//MeV, average neutron energy
  const Double_t EnRes = 50.;//MeV, energy spread of neutron(FWHM)
  const Double_t Eg0 = 1;//MeV, gamma energy  
  const Double_t Rg = 0.3;//ratio of gamma,ratio of neutron 1-Rg 

  //1. treeBranch
  Double_t x;//
  Double_t e;//
  int pid;    //n:pid=1,g:pid=0
  Double_t tof, ctof;//TOF:cTOFTOF
  Double_t tu, td;
  Double_t qu, qd;

  Double_t tu_off = 5.5;//time offset//+PMT+
  Double_t td_off = 20.4;//time offset 

 //2. ROOTTree 
  TFile *opf = new TFile("tree.root", "recreate");//tree.root *opf
  TTree *opt = new TTree("tree", "tree structure");//tree *opt

  //3. tree
    //DDouble_tI Int_t
  opt->Branch("x", &x, "x/D");
  opt->Branch("e", &e, "e/D");
  opt->Branch("tof", &tof, "tof/D");
  opt->Branch("ctof",&ctof,"ctof/D");
  opt->Branch("pid", &pid, "pid/I");
  opt->Branch("tu", &tu, "tu/D");
  opt->Branch("td", &td, "td/D");
  opt->Branch("qu", &qu, "qu/D"); 
  opt->Branch("qd", &qd, "qd/D");  

// histogramROOTTTreehistogramgraph
  TH1D *hctof = new TH1D("hctof","neutron time of flight",1000,0,100);
  TRandom3 *gr = new TRandom3(0);//

  //4. tree
  for(int i=0; i<100000; i++){
    x = gr->Uniform(-L, L);//-L,L.
    Double_t Dr = D + gr->Uniform(-0.5, 0.5) * dD;//
    Double_t d = TMath::Sqrt(Dr * Dr + x * x);//flight path,//sqrt(),abs();TMath::Abs()
    if(gr->Uniform() < Rg) { //gamma
       pid = 0;
       e = Eg0;
       tof = 3.333 * (d * 0.01);
    }
    else {  //neutron
        pid = 1;
        e = gr->Gaus(En0, EnRes/2.35); // energy of neutrons
        tof = 72.29824/TMath::Sqrt(e) * (d * 0.01);//ns
    }
    tu = tof + (L - x)/Vsc+gr->Gaus(0,TRes/2.35)+tu_off;
    td = tof + (L + x)/Vsc+gr->Gaus(0,TRes/2.35)+td_off;
    ctof = (tu + td)/2.;//simplified calculation.
    hctof->Fill(ctof);
//neutronenergy of recoil proton in plas. q0=0-En gammaq0=0-Egammacompton plateau
    Double_t q0 = e * gr->Uniform(); //light response of Qee = f(Ep) is not considered
    qu = q0 *TMath::Exp(-(L - x)/Lambda);
    qu = gr->Gaus(qu, qu * QRes/2.35);//QRes, relative energy resolution
    qd = q0 * TMath::Exp(-(L + x)/Lambda);
    qd = gr->Gaus(qd, qd * QRes/2.35);
    opt->Fill();//5.Tree

    if(i%1000 == 0) cout << ".";
  }
  cout << endl;
  // 6.root
  hctof->Write();//histogram
  opt->Write();//TTree
  opf->Close();//
}
