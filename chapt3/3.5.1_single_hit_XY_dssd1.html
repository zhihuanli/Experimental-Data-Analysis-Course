bool IsSingleHit(int dxe[32], int dye[32], int ix, int iy) {
    bool bxy, bxl, bxr, byl, byr;

    bxy = dxe[ix] > 200 && dxe[ix] < 8000 && 
          dye[iy] > 200 && dye[iy] < 8000;
    bxl = (ix==0) ? 1 : dxe[ix-1] <50;  //cross-talk threshold
    bxr = (ix==31) ? 1 : dxe[ix+1] <50;
    byl = (iy==0) ? 1 : dye[iy-1] <50;
    byr = (iy==31) ? 1 : dye[iy+1] <50;    

    return bxy && bxl && bxr && byl && byr;
}

for (Long64_t jentry = 0; jentry < nentries; jentry++) {
    tree->GetEntry(jentry);
    bool bcut;
    Int_t *dx[3] = {d1x,d2x,d3x};
    Int_t *dy[3] = {d1y,d2y,d3y};
    Int_t did = id -1;
    for (int i = 0; i < 32; i++) {
        for (int j = 0; j < 32; j++) {
            bcut = cutxy[did]->IsInside(dx[did][i], dy[did][j]);
            bcut *= IsSingleHit(dx[did],dy[did],i,j);
            if (!bcut) continue;
            ix = i;
            iy = j;
            xe = dx[did][i];
            ye = dy[did][j];
            tout->Fill();

        }
    }
    if (jentry % 10000 == 0 ) cout << ".";

}
