gFit¶

  • IsXY = "X" for fitting g(x,y) with y=k*x+b
  • IsXY = "Y" for fitting g(x,y) with x=k*x+b
  • par[3] for TF2
  • stat[3]: ymin,ymax,Ncout for fitting
  • minCount: minimum count for TGraph fitting
In [12]:
bool gFit(char IsXY, TGraph *g, double* par, double &chi2ndf, int *sta,int minCount=20)
{    
    //cout<<g->GetN()<<endl;
    if(g->GetN() < minCount)
        return 0;
    TGraph *g1=new TGraph;
    TGraph *g2=new TGraph;
    TF1 *fpa1= new TF1("fp1","pol1",200,8000);
    TF1 *fpa2= new TF1("fp2","pol2",200,8000);
    int ymin=10000, ymax=0;
    double par1[2];
    for(int k=0; k<g->GetN(); k++) {
            double x=g->GetX()[k];
            double y=g->GetY()[k];
            if(IsXY=='Y' or IsXY=='y') swap(x,y);//x=ky+b
            g1->SetPoint(k,x,y);
    }
    g1->Fit(fpa1,"Q ROB");
    fpa1->GetParameters(&par1[0]);
    int ng = 0;
    for(int k=0; k<g1->GetN(); k++) {
            double x=g1->GetX()[k];
            double y=g1->GetY()[k];
            if (abs(y-(par1[0]+par1[1]*x))<20) { //eliminate outliers
                g2->SetPoint(ng,x,y);
                ng++;
            }           
    }
    if(g2->GetN()>= minCount) {
        g2->Fit(fpa2,"Q ROB");
        fpa2->GetParameters(&par[0]);
        chi2ndf=fpa2->GetChisquare()/fpa2->GetNDF();
        for(int j = 0;j < g2->GetN(); j++){
            double x=g2->GetX()[j];
            double y=g2->GetY()[j];
            if(y>ymax) ymax=y;
            if(y<ymin) ymin=y;
        }
    }
    sta[0]=ymin;
    sta[1]=ymax;
    sta[2]=g2->GetN();
    delete fpa1;
    delete fpa2;
    delete g1;
    delete g2;
    if(sta[2] < minCount) 
        return 0;
    else 
        return 1;    
}

CalibMerge¶

  • Calibrate & Merge xstrips
  • IsXY ='X' merge xstrips within the range of m1 to m2
    • ic: ystrip
    • gc: merged & calibrated graph for gr[m1-m2][ic]
  • IsXY ='Y' merge ystrips within the range of m1 to m2
    • ic: xstrip
    • gc: merged & calibrated graph for gr[ic][m1-m2]
In [15]:
void CalibMerge(char IsXY,int m1,int m2, int ic, TGraph *gc)
{
    gc->Clear();
    int ngy=0;
    if(IsXY=='X') {
        for(int ix=m1;ix<=m2;ix++) {
            for(int i=0;i<gr[ix][ic]->GetN();i++) {
                double x=gr[ix][ic]->GetX()[i];
                double y=gr[ix][ic]->GetY()[i];
                x = parx[ix][0]+parx[ix][1]*x+parx[ix][2]*x*x;
                gc->SetPoint(ngy,x,y);
                ngy++;
            }
        }
    }
    else {
        for(int iy=m1;iy<=m2;iy++) {
            for(int i=0;i<gr[ic][iy]->GetN();i++) {
                double x=gr[ic][iy]->GetX()[i];
                double y=gr[ic][iy]->GetY()[i];
                y = pary[iy][0]+pary[iy][1]*y+pary[iy][2]*y*y;
                gc->SetPoint(ngy,x,y);
                ngy++;
            }
        }        
    }
}
