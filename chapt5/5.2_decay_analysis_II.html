重离子与衰变事件的关联¶

位置关联¶

  • DSSD()DSSDpixel()DSSD $x \times y$ ht,hx,hy; bt,be,bx,by

时间关联¶

  • (ht)(bt)-$(ht_i,bt_j),i \neq j$

  • (.a)

    • ()
        1. (ht1,bt1)
        1. (ht2,bt2)
        1. (ht3,bt3)

  • $\delta t=bt-ht$

  • (.b)

    • ()
        1. (ht1,bt1
        1. (ht2,bt2),(ht2,bt4)
        1. (ht3,bt2),(ht3,bt4),(ht3,bt3)
        1. (ht4,bt4),(ht4,bt3),(ht4,bt5

  • $(ht_i,bt_j),i \neq j$ $\delta t=bt-ht$

  • $(ht_i,bt_j),i = j$ $\delta t=bt-ht$

  • position-time correlation for implantation and decay events

1.位置关联条件¶

  • :hx==bx && hy==by, hz==bz(dssd);

  • (beta):abs(hx-bx)<2 && abs(hx-bx)<2;

    • $\alpha$ $\Delta x/y = 0$

ULong64_t twindow = 20000000000;//20s
TH2F *hedt=new TH2F("hedt","decayenergy vs decaytime in the same pixiel",200,0,2e10,300,5000,8000);
TH1F *hdt=new TH1F("hdt","decaytime in the same pixiel",200,0,2e10);
for(auto ia=mapimp.begin(); ia!=mapimp.end();ia++) {//multimap<ULong64_t,dssd>::iterator ia=mapimap.begin(), auto ia=
     auto ib=mapdec.lower_bound(ia->first);
     for(auto ic=ib; ic!=mapdec.end();ic++) {
        if(ic->first >= ia->first + twindow) break;
        Int_t delx=abs(ic->second.xstrip-ia->second.xstrip);
        Int_t dely=abs(ic->second.ystrip-ia->second.ystrip);         
        Long64_t decaytime = (ic->first - ia->first);
        Double_t e=ic->second.energy;
        if(delx==0 && dely==0) {
            hedt->Fill(decaytime,e);//same pixel (0,0)
            if(e>7400 && e<7510) hdt->Fill(decaytime);
        }
    }   
}

twindow = 20000000000;//20s
TH2F *hbe2=new TH2F("hbe2","de_0:de_2",200,6000,8000,200,6000,8000);
TH1F *ht2=new TH1F("ht2","decay time of daughter",200,0,2e10);
for(auto ia=mapimp.begin(); ia!=mapimp.end();ia++) {
     ULong64_t ht=ia->first; 
     auto ib=mapdec.lower_bound(ia->first);
     Double_t be[10];
     Long64_t bt[10];
     int bhit=0;
     for(auto ic=ib; ic!=mapdec.end();ic++) {
        if(ic->first >= ia->first + twindow) break;
        Int_t delx=abs(ic->second.xstrip-ia->second.xstrip);
        Int_t dely=abs(ic->second.ystrip-ia->second.ystrip);         
        if(delx!=0 || dely!=0) continue; //
        be[bhit]=ic->second.energy;
        bt[bhit]=ic->first;
        bhit++;
        if(bhit>10) break;    
    }   
    if(bhit>1 && bt[0]-ht<700000000 && bt[1]-bt[0]<19000000000) {
        hbe2->Fill(be[0],be[1]);
        ht2->Fill(bt[1]-bt[0]);
    }
}
